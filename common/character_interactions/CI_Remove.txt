	icon = "gfx/interface/character_panel/grant_command.dds"
	potential = {
		NOT = { has_variable = openCI_Remove_var }
	}

	ai_chance = {
		base = 0
    }

	effect = {
		if = {
			limit = {
				NOT = { has_variable = openCI_Remove_var }

		}

			remove_variable = openCI_Remove_var
		}
	}

}
